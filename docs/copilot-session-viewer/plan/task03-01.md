```typescript
// src/lib/ws-terminal.ts

import type { IncomingMessage } from "http";
import { getActiveSessions } from "./terminal";

// === 型定義 ===
export interface InputMessage { type: "input"; data: string; }
export interface ResizeMessage { type: "resize"; cols: number; rows: number; }
export interface PingMessage { type: "ping"; }
export type ClientMessage = InputMessage | ResizeMessage | PingMessage;

export interface ConnectedMessage {
  type: "connected"; sessionId: string; tmuxPane: string;
  containerId?: string; cols: number; rows: number;
}
export interface OutputMessage { type: "output"; data: string; }
export interface ErrorMessage { type: "error"; code: ErrorCode; message: string; }
export interface PongMessage { type: "pong"; }
export interface DisconnectedMessage { type: "disconnected"; reason: DisconnectReason; }
export type ServerMessage = ConnectedMessage | OutputMessage | ErrorMessage | PongMessage | DisconnectedMessage;

export type ErrorCode = "SESSION_NOT_FOUND" | "PANE_NOT_FOUND" | "CAPTURE_FAILED" | "AUTH_FAILED" | "CONNECTION_LIMIT";
export type DisconnectReason = "session_ended" | "pane_closed" | "timeout" | "capture_failed" | "auth_expired" | "server_shutdown";

export interface ResolvedSession {
  sessionId: string; tmuxPane: string; containerId?: string; containerUser?: string;
}

// === 認証 ===
export function authenticateUpgrade(request: IncomingMessage): {
  authenticated: boolean; reason?: "no_auth_config" | "invalid_credentials" | "missing_header";
} {
  const user = process.env.BASIC_AUTH_USER;
  const pass = process.env.BASIC_AUTH_PASS;
  if (!user || !pass) return { authenticated: false, reason: "no_auth_config" };

  const authHeader = request.headers.authorization;
  if (!authHeader) return { authenticated: false, reason: "missing_header" };

  const [scheme, encoded] = authHeader.split(" ");
  if (scheme !== "Basic" || !encoded) return { authenticated: false, reason: "invalid_credentials" };

  const decoded = Buffer.from(encoded, "base64").toString();
  const [u, p] = decoded.split(":");
  if (u === user && p === pass) return { authenticated: true };
  return { authenticated: false, reason: "invalid_credentials" };
}

// === セッション解決 ===
export async function resolveSession(sessionId: string): Promise<ResolvedSession | null> {
  const sessions = await getActiveSessions();
  const session = sessions.find((s) => s.id === sessionId);
  if (!session || !session.tmuxPane) return null;
  return {
    sessionId: session.id,
    tmuxPane: session.tmuxPane,
    containerId: session.containerId,
    containerUser: session.containerUser,
  };
}
```

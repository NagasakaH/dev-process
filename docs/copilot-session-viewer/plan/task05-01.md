### 実装ステップ

1. **hooks ディレクトリ作成**
   ```bash
   mkdir -p src/hooks
   ```

2. **型の再利用**
   - ws-terminal.ts からフロントエンド用の型を分離（ConnectionState, UseTerminalWebSocketReturn, UseTerminalWebSocketOptions）
   - または共有型ファイルを作成

3. **useTerminalWebSocket フック実装**
   - `useState` で connectionState 管理
   - `useRef` で WebSocket インスタンス管理
   - `useEffect` で接続のライフサイクル管理
   - WebSocket URL 構築: `ws://${location.host}/ws/terminal?sessionId=${sessionId}`
   - メッセージハンドラー:
     - `connected` → connectionState = "connected", onConnected
     - `output` → onOutput
     - `error` → connectionState = "error", onError
     - `disconnected` → connectionState = "disconnected", onDisconnected
     - `pong` → KeepAlive 応答受信
   - 再接続ロジック: 1秒→2秒→4秒（指数バックオフ、最大3回）
   - `sendInput`: WebSocket 経由で InputMessage 送信
   - `sendResize`: WebSocket 経由で ResizeMessage 送信
   - `disconnect`: WebSocket close + クリーンアップ

4. **KeepAlive ping**
   - 30秒間隔で `{ type: "ping" }` 送信
   - pong 応答タイムアウト（15秒）で再接続

### GREEN: 最小限の実装

```typescript
// src/hooks/useTerminalWebSocket.ts

import { useState, useRef, useEffect, useCallback } from "react";
import type { ConnectedMessage, ErrorMessage } from "../lib/ws-terminal";

export type ConnectionState = "connecting" | "connected" | "reconnecting" | "disconnected" | "error";

export interface UseTerminalWebSocketOptions {
  onOutput?: (data: string) => void;
  onConnected?: (info: ConnectedMessage) => void;
  onError?: (error: ErrorMessage) => void;
  onDisconnected?: (reason: string) => void;
}

export function useTerminalWebSocket(
  sessionId: string | null,
  options?: UseTerminalWebSocketOptions
) {
  const [connectionState, setConnectionState] = useState<ConnectionState>("disconnected");
  const [error, setError] = useState<string | null>(null);
  const wsRef = useRef<WebSocket | null>(null);
  const reconnectCountRef = useRef(0);

  const connect = useCallback(() => {
    if (!sessionId) return;
    const ws = new WebSocket(`ws://${location.host}/ws/terminal?sessionId=${sessionId}`);
    wsRef.current = ws;
    setConnectionState("connecting");

    ws.onopen = () => { /* wait for connected message */ };
    ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);
      switch (msg.type) {
        case "connected": setConnectionState("connected"); options?.onConnected?.(msg); break;
        case "output": options?.onOutput?.(msg.data); break;
        case "error": setConnectionState("error"); setError(msg.message); options?.onError?.(msg); break;
        case "disconnected": setConnectionState("disconnected"); options?.onDisconnected?.(msg.reason); break;
        case "pong": break;
      }
    };
    ws.onclose = () => { /* reconnect logic */ };
  }, [sessionId, options]);

  useEffect(() => { connect(); return () => wsRef.current?.close(); }, [connect]);

  const sendInput = useCallback((data: string) => {
    wsRef.current?.send(JSON.stringify({ type: "input", data }));
  }, []);

  const sendResize = useCallback((cols: number, rows: number) => {
    wsRef.current?.send(JSON.stringify({ type: "resize", cols, rows }));
  }, []);

  const disconnect = useCallback(() => {
    wsRef.current?.close();
    setConnectionState("disconnected");
  }, []);

  return { connectionState, error, sendInput, sendResize, disconnect };
}
```

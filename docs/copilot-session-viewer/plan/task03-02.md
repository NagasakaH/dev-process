```typescript
// src/lib/__tests__/ws-terminal.test.ts に追加

import { capturePane, sendInput, SPECIAL_KEY_MAP } from "../ws-terminal";
import { execFileSync } from "child_process";

vi.mock("child_process", async (importOriginal) => {
  const actual = await importOriginal<typeof import("child_process")>();
  return {
    ...actual,
    execFileSync: vi.fn(),
  };
});

describe("capturePane", () => {
  const mockExecFileSync = vi.mocked(execFileSync);

  beforeEach(() => {
    mockExecFileSync.mockReset();
  });

  // UT-7: ローカル tmux capture-pane
  it("ローカル tmux で capture-pane -p -e を実行し ANSI 付き出力を返す", () => {
    mockExecFileSync.mockReturnValue(Buffer.from("\x1b[32mtest\x1b[0m"));
    const result = capturePane("0:1.0");
    expect(result).toContain("\x1b[32mtest\x1b[0m");
    expect(result).toMatch(/^\x1b\[H\x1b\[2J/); // MRD-002: クリアシーケンスプリペンド
  });

  // UT-8: Docker exec 経由
  it("Docker exec 経由で capture-pane を実行", () => {
    mockExecFileSync.mockReturnValue(Buffer.from("docker output"));
    const result = capturePane("0:1.0", "container123", "1000");
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "docker",
      expect.arrayContaining(["exec", "-u", "1000", "container123", "tmux"]),
      expect.any(Object)
    );
    expect(result).toContain("docker output");
  });

  // UT-9: 失敗時空文字列
  it("capture-pane 失敗時に空文字列を返す", () => {
    mockExecFileSync.mockImplementation(() => { throw new Error("tmux error"); });
    const result = capturePane("0:1.0");
    expect(result).toBe("");
  });
});

describe("sendInput", () => {
  const mockExecFileSync = vi.mocked(execFileSync);

  beforeEach(() => {
    mockExecFileSync.mockReset();
  });

  // UT-10: 通常文字
  it("通常文字を send-keys -l で送信", () => {
    sendInput("0:1.0", "hello");
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["send-keys", "-t", "0:1.0", "-l", "hello"]),
      expect.any(Object)
    );
  });

  // UT-11: Enter キー
  it("\\r を tmux Enter に変換", () => {
    sendInput("0:1.0", "\r");
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["send-keys", "-t", "0:1.0", "Enter"]),
      expect.any(Object)
    );
  });

  // UT-12: Ctrl+C
  it("\\x03 を tmux C-c に変換", () => {
    sendInput("0:1.0", "\x03");
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["send-keys", "-t", "0:1.0", "C-c"]),
      expect.any(Object)
    );
  });

  // UT-13: 矢印キー
  it("\\x1b[A を tmux Up に変換", () => {
    sendInput("0:1.0", "\x1b[A");
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["send-keys", "-t", "0:1.0", "Up"]),
      expect.any(Object)
    );
  });

  // UT-14: 混合入力
  it("文字+Enter を分割して送信", () => {
    sendInput("0:1.0", "ls\r");
    expect(mockExecFileSync).toHaveBeenCalledTimes(2);
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["-l", "ls"]),
      expect.any(Object)
    );
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["Enter"]),
      expect.any(Object)
    );
  });

  // UT-15: Docker exec 経由
  it("Docker exec 経由で send-keys を送信", () => {
    sendInput("0:1.0", "a", "container123", "1000");
    expect(mockExecFileSync).toHaveBeenCalledWith(
      "docker",
      expect.arrayContaining(["exec", "-u", "1000", "container123", "tmux"]),
      expect.any(Object)
    );
  });
});

describe("差分検出", () => {
  // UT-17: 同一出力→スキップ
  it("前回と同じ出力の場合は送信しない（shouldSendOutput = false）", () => {
    // shouldSendOutput(current, last) ユーティリティのテスト
    const { shouldSendOutput } = require("../ws-terminal");
    expect(shouldSendOutput("same output", "same output")).toBe(false);
  });

  // UT-18: 変更あり→送信
  it("前回と異なる出力の場合は送信する（shouldSendOutput = true）", () => {
    const { shouldSendOutput } = require("../ws-terminal");
    expect(shouldSendOutput("new output", "old output")).toBe(true);
  });
});

describe("SPECIAL_KEY_MAP", () => {
  // UT-20: 全マッピング
  it("全必須キーマッピングが定義されている", () => {
    const requiredKeys = ["\r", "\t", "\x7f", "\x1b", "\x03", "\x04", "\x1a",
      "\x0c", "\x01", "\x05", "\x0b", "\x15", "\x17",
      "\x1b[A", "\x1b[B", "\x1b[C", "\x1b[D", "\x1b[H", "\x1b[F",
      "\x1b[2~", "\x1b[3~", "\x1b[5~", "\x1b[6~"];
    for (const key of requiredKeys) {
      expect(SPECIAL_KEY_MAP).toHaveProperty(key);
    }
  });
});
```

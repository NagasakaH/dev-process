### RED: 失敗するテストケース

```typescript
// e2e/terminal-viewer.spec.ts

import { test, expect } from "@playwright/test";

test.describe("ターミナルビューア: 基本フロー", () => {

  // E2E-1: ターミナルモーダル表示
  test("セッションカードのターミナルボタンクリックでモーダルが表示される", async ({ page }) => {
    await page.goto("/");

    // アクティブセッションが表示されるまで待機
    await page.waitForSelector("[data-testid='active-sessions']");

    // ターミナルボタンをクリック
    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await expect(terminalBtn).toBeVisible();
    await terminalBtn.click();

    // モーダルが表示される
    const modal = page.locator("[role='dialog']");
    await expect(modal).toBeVisible();

    // xterm.js のターミナル領域が見える
    const terminal = page.locator(".xterm");
    await expect(terminal).toBeVisible();
  });

  // E2E-2: リアルタイム表示
  test("ターミナルモーダルに tmux pane の内容がリアルタイム表示される", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await terminalBtn.click();

    // xterm.js 領域にテキストが表示される（プロンプトなど）
    const terminal = page.locator(".xterm-screen");
    await expect(terminal).toBeVisible();

    // 数秒待って内容が更新されることを確認
    await page.waitForTimeout(2000);
    const text = await terminal.textContent();
    expect(text).toBeTruthy(); // 何かしらのテキストが表示されている
  });

  // E2E-3: キー入力送信
  test("キーボード入力が tmux pane に送信される", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await terminalBtn.click();

    const terminal = page.locator(".xterm");
    await expect(terminal).toBeVisible();

    // "echo hello" + Enter を入力
    await terminal.click(); // フォーカス
    await page.keyboard.type("echo hello");
    await page.keyboard.press("Enter");

    // tmux pane に "hello" が表示されるまで待機
    await page.waitForTimeout(3000);
    const screen = page.locator(".xterm-screen");
    const text = await screen.textContent();
    expect(text).toContain("hello");
  });

  // E2E-4: モーダル閉じる
  test("✕ボタンでモーダルが閉じる", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await terminalBtn.click();

    const modal = page.locator("[role='dialog']");
    await expect(modal).toBeVisible();

    // ✕ボタンクリック
    const closeBtn = page.locator("[role='dialog'] button").filter({ hasText: /✕|×|close/i }).first();
    await closeBtn.click();

    // モーダルが消える
    await expect(modal).not.toBeVisible();

    // セッション一覧が見える
    await expect(page.locator("[data-testid='active-sessions']")).toBeVisible();
  });

  // E2E-5: Docker コンテナ内セッション
  test("Docker コンテナ内セッションのターミナルビューが動作する", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    // Docker コンテナ内のセッション（containerId あり）のターミナルボタンをクリック
    // コンテナ内の tmux pane 内容が表示されることを確認
    const terminalBtns = page.locator("button[title='ターミナルを開く']");
    const count = await terminalBtns.count();
    expect(count).toBeGreaterThan(0);

    // 最後のボタン（Docker セッション想定）をクリック
    await terminalBtns.last().click();

    const terminal = page.locator(".xterm-screen");
    await expect(terminal).toBeVisible();
    await page.waitForTimeout(2000);
    const text = await terminal.textContent();
    expect(text).toBeTruthy();
  });
});
```

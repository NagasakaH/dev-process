describe("接続数制限", () => {
  beforeEach(() => { connections.clear(); });

  // UT-19: 同一pane制限
  it("同一 pane に MAX_CONNECTIONS_PER_PANE 超の接続で拒否", () => {
    // 2つの接続を追加
    const ws1 = {} as any;
    const ws2 = {} as any;
    connections.set(ws1, { tmuxPane: "0:1.0", sessionId: "s1" } as any);
    connections.set(ws2, { tmuxPane: "0:1.0", sessionId: "s2" } as any);

### GREEN: 最小限の実装

```typescript
// getPaneSize
export function getPaneSize(tmuxPane: string, containerId?: string, containerUser?: string) {
  const args = containerId
    ? ["exec", ...(containerUser ? ["-u", containerUser] : []), containerId, "tmux", "list-panes", "-t", tmuxPane, "-F", "#{pane_width} #{pane_height}"]
    : ["list-panes", "-t", tmuxPane, "-F", "#{pane_width} #{pane_height}"];
  const cmd = containerId ? "docker" : "tmux";
  const output = execFileSync(cmd, args, { encoding: "utf-8" }).trim();
  const [cols, rows] = output.split(" ").map(Number);
  return { cols, rows };
}

// resizePane
export function resizePane(tmuxPane: string, cols: number, rows: number, containerId?: string, containerUser?: string) {
  if (cols < 1 || rows < 1) return;
  const args = containerId
    ? ["exec", ...(containerUser ? ["-u", containerUser] : []), containerId, "tmux", "resize-pane", "-t", tmuxPane, "-x", String(cols), "-y", String(rows)]
    : ["resize-pane", "-t", tmuxPane, "-x", String(cols), "-y", String(rows)];
  execFileSync(containerId ? "docker" : "tmux", args, { encoding: "utf-8" });
}

// canConnect
export function canConnect(tmuxPane: string, isDocker: boolean): boolean {
  const paneCount = [...connections.values()].filter(c => c.tmuxPane === tmuxPane).length;
  if (paneCount >= MAX_CONNECTIONS_PER_PANE) return false;
  const totalMax = isDocker ? MAX_TOTAL_CONNECTIONS_DOCKER : MAX_TOTAL_CONNECTIONS_LOCAL;
  if (connections.size >= totalMax) return false;
  return true;
}
```

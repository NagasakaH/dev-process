// src/lib/__tests__/ws-terminal.test.ts - 結合テスト部分

import { WebSocket, WebSocketServer } from "ws";
import http from "http";
import { setupTerminalWebSocket } from "../ws-terminal";

describe("結合テスト: WebSocket ターミナル", () => {
  let server: http.Server;
  let port: number;

  beforeAll(async () => {
    server = http.createServer();
    setupTerminalWebSocket(server);
    await new Promise<void>((resolve) => {
      server.listen(0, () => {
        port = (server.address() as any).port;
        resolve();
      });
    });
  });

  afterAll(() => {
    server.close();
  });

  // IT-1: WS接続→connected メッセージ
  it("WebSocket 接続確立後に connected メッセージが返る", async () => {
    process.env.BASIC_AUTH_USER = "admin";
    process.env.BASIC_AUTH_PASS = "secret";
    const auth = Buffer.from("admin:secret").toString("base64");
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );

    const msg = await new Promise<any>((resolve) => {
      ws.on("message", (data) => resolve(JSON.parse(data.toString())));
    });

    expect(msg.type).toBe("connected");
    expect(msg.sessionId).toBe("session-001");
    ws.close();
  });

  // IT-2: capture-pane→WS出力
  it("capture-pane の結果が output メッセージとして送信される", async () => {
    const auth = Buffer.from("admin:secret").toString("base64");
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );

    const messages: any[] = [];
    ws.on("message", (data) => messages.push(JSON.parse(data.toString())));

    await new Promise((r) => setTimeout(r, 500)); // capture ループ待ち
    const outputMsg = messages.find((m) => m.type === "output");
    expect(outputMsg).toBeDefined();
    expect(outputMsg.data).toContain("\x1b[H\x1b[2J"); // MRD-002
    ws.close();
  });

  // IT-3: WS入力→send-keys
  it("InputMessage 受信後に send-keys が実行される", async () => {
    const auth = Buffer.from("admin:secret").toString("base64");
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );

    await new Promise<void>((resolve) => ws.on("open", resolve));
    ws.send(JSON.stringify({ type: "input", data: "ls\r" }));
    await new Promise((r) => setTimeout(r, 100));

    // sendInput が呼ばれたことを mock で検証
    expect(vi.mocked(execFileSync)).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["send-keys"]),
      expect.any(Object)
    );
    ws.close();
  });

  // IT-4: 認証→WS接続拒否
  it("認証失敗時に接続が拒否される", async () => {
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: "Basic " + Buffer.from("wrong:wrong").toString("base64") } }
    );

    await new Promise<void>((resolve) => {
      ws.on("error", () => resolve());
      ws.on("close", () => resolve());
    });
    expect(ws.readyState).not.toBe(WebSocket.OPEN);
  });

  // IT-5: WS切断→クリーンアップ
  it("WebSocket close 後にキャプチャループが停止し接続ストアから削除", async () => {
    const auth = Buffer.from("admin:secret").toString("base64");
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );
    await new Promise<void>((resolve) => ws.on("open", resolve));

    const connCountBefore = connections.size;
    ws.close();
    await new Promise((r) => setTimeout(r, 200));
    expect(connections.size).toBeLessThan(connCountBefore);
  });

  // IT-6: Docker環境 capture + send
  it("containerId ありの場合 docker exec 経由で capture-pane と send-keys が実行される", async () => {
    // Docker セッションのモック設定
    const mockGetActiveSessions = vi.mocked(getActiveSessions);
    mockGetActiveSessions.mockReturnValue([
      {
        sessionId: "docker-session-001",
        tmuxPane: "0:1.0",
        containerId: "abc123def456",
        containerUser: "1000",
        status: "active",
      },
    ]);

    const auth = Buffer.from("admin:secret").toString("base64");
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=docker-session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );

    // connected メッセージを待機
    const connectedMsg = await new Promise<any>((resolve) => {
      ws.on("message", (data) => {
        const msg = JSON.parse(data.toString());
        if (msg.type === "connected") resolve(msg);
      });
    });
    expect(connectedMsg.sessionId).toBe("docker-session-001");

    // capture-pane が docker exec 経由で呼ばれたことを検証
    await new Promise((r) => setTimeout(r, 500));
    expect(vi.mocked(execFileSync)).toHaveBeenCalledWith(
      "docker",
      expect.arrayContaining(["exec", "-u", "1000", "abc123def456", "tmux", "capture-pane"]),
      expect.any(Object)
    );

    // send-keys も docker exec 経由で呼ばれることを検証
    ws.send(JSON.stringify({ type: "input", data: "pwd\r" }));
    await new Promise((r) => setTimeout(r, 100));
    expect(vi.mocked(execFileSync)).toHaveBeenCalledWith(
      "docker",
      expect.arrayContaining(["exec", "-u", "1000", "abc123def456", "tmux", "send-keys"]),
      expect.any(Object)
    );

    ws.close();
  });

  // IT-7: 接続再確立
  it("一度切断後に再接続で新しい connected メッセージを受信できる", async () => {
    const auth = Buffer.from("admin:secret").toString("base64");

    // 1回目の接続
    const ws1 = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );
    const msg1 = await new Promise<any>((resolve) => {
      ws1.on("message", (data) => resolve(JSON.parse(data.toString())));
    });
    expect(msg1.type).toBe("connected");

    // 切断
    ws1.close();
    await new Promise((r) => setTimeout(r, 300));

    // 2回目の接続
    const ws2 = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );
    const msg2 = await new Promise<any>((resolve) => {
      ws2.on("message", (data) => resolve(JSON.parse(data.toString())));
    });
    expect(msg2.type).toBe("connected");
    expect(msg2.sessionId).toBe("session-001");

    // 切断後の再接続で新しい capture ループが開始されること
    const messages: any[] = [];
    ws2.on("message", (data) => messages.push(JSON.parse(data.toString())));
    await new Promise((r) => setTimeout(r, 500));
    const outputMsg = messages.find((m) => m.type === "output");
    expect(outputMsg).toBeDefined();

    ws2.close();
  });

  // IT-8: resize メッセージ→tmux resize-pane
  it("ResizeMessage 受信後に tmux resize-pane が実行される", async () => {
    const auth = Buffer.from("admin:secret").toString("base64");
    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`,
      { headers: { authorization: `Basic ${auth}` } }
    );
    await new Promise<void>((resolve) => ws.on("open", resolve));
    ws.send(JSON.stringify({ type: "resize", cols: 120, rows: 40 }));
    await new Promise((r) => setTimeout(r, 100));

    expect(vi.mocked(execFileSync)).toHaveBeenCalledWith(
      "tmux",
      expect.arrayContaining(["resize-pane", "-x", "120", "-y", "40"]),
      expect.any(Object)
    );
    ws.close();
  });

  // IT-9: 認証未設定時の接続拒否
  it("BASIC_AUTH 未設定時に 403 で接続拒否", async () => {
    delete process.env.BASIC_AUTH_USER;
    delete process.env.BASIC_AUTH_PASS;

    const ws = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-001`
    );

    await new Promise<void>((resolve) => {
      ws.on("error", () => resolve());
      ws.on("close", () => resolve());
    });
    expect(ws.readyState).not.toBe(WebSocket.OPEN);
  });

  // IT-10: ローカル環境の総接続上限
  it("ローカル環境で6番目のWebSocket接続が拒否される", async () => {
    process.env.BASIC_AUTH_USER = "admin";
    process.env.BASIC_AUTH_PASS = "secret";
    const auth = Buffer.from("admin:secret").toString("base64");

    const clients: WebSocket[] = [];
    for (let i = 0; i < 5; i++) {
      const ws = new WebSocket(
        `ws://localhost:${port}/ws/terminal?sessionId=session-${i}`,
        { headers: { authorization: `Basic ${auth}` } }
      );
      await new Promise<void>((resolve) => ws.on("open", resolve));
      clients.push(ws);
    }

    const ws6 = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=session-5`,
      { headers: { authorization: `Basic ${auth}` } }
    );
    await new Promise<void>((resolve) => {
      ws6.on("error", () => resolve());
      ws6.on("close", () => resolve());
    });
    expect(ws6.readyState).not.toBe(WebSocket.OPEN);

    clients.forEach((c) => c.close());
  });

  // IT-11: Docker環境の総接続上限
  it("Docker環境で3番目のWebSocket接続が拒否される", async () => {
    // Docker セッションのモック設定（複数セッション）
    const mockGetActiveSessions = vi.mocked(getActiveSessions);
    mockGetActiveSessions.mockReturnValue(
      Array.from({ length: 3 }, (_, i) => ({
        sessionId: `docker-limit-${i}`,
        tmuxPane: `0:${i}.0`,
        containerId: "container-limit-test",
        containerUser: "1000",
        status: "active",
      }))
    );

    const auth = Buffer.from("admin:secret").toString("base64");
    const clients: WebSocket[] = [];

    // Docker 環境の上限は 2 接続
    for (let i = 0; i < 2; i++) {
      const ws = new WebSocket(
        `ws://localhost:${port}/ws/terminal?sessionId=docker-limit-${i}`,
        { headers: { authorization: `Basic ${auth}` } }
      );
      await new Promise<void>((resolve) => ws.on("open", resolve));
      clients.push(ws);
    }

    // 3番目の接続は拒否される
    const ws3 = new WebSocket(
      `ws://localhost:${port}/ws/terminal?sessionId=docker-limit-2`,
      { headers: { authorization: `Basic ${auth}` } }
    );
    await new Promise<void>((resolve) => {
      ws3.on("error", () => resolve());
      ws3.on("close", () => resolve());
    });
    expect(ws3.readyState).not.toBe(WebSocket.OPEN);

    clients.forEach((c) => c.close());
  });
});

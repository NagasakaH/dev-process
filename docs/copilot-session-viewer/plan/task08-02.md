### RED: 失敗するテストケース

```typescript
// e2e/terminal-viewer-auth.spec.ts

test.describe("ターミナルビューア: 既存機能回帰", () => {
  // E2E-6: セッション一覧
  test("既存のセッション一覧が正常に動作する", async ({ page }) => {
    await page.goto("/");
    const sessions = page.locator("[data-testid='active-sessions']");
    await expect(sessions).toBeVisible();

    // セッションカードが1つ以上表示
    const cards = sessions.locator("[data-testid='session-card']");
    await expect(cards.first()).toBeVisible();
  });

  // E2E-7: ask_user 応答
  test("ask_user 応答機能が正常に動作する", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    // ask_user 待ちセッションがある場合、応答入力欄が存在
    const askUserInput = page.locator("[data-testid='ask-user-input']");
    if (await askUserInput.isVisible()) {
      await askUserInput.fill("test response");
      const submitBtn = page.locator("[data-testid='ask-user-submit']");
      await submitBtn.click();
      // エラーが発生しないことを確認
    }
  });
});

test.describe("ターミナルビューア: 認証", () => {
  // E2E-8: 認証付き WS 接続
  test("Basic Auth 認証後にターミナルモーダルが動作する", async ({ page }) => {
    await page.goto("/");
    // Basic Auth はブラウザが自動処理（Playwright context で設定済み）

    await page.waitForSelector("[data-testid='active-sessions']");
    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await terminalBtn.click();

    const terminal = page.locator(".xterm");
    await expect(terminal).toBeVisible();
  });

  // E2E-9: 未認証 WS 接続拒否
  test("認証なしで /ws/terminal に WebSocket 接続が拒否される", async ({ page }) => {
    // 認証なしの WebSocket 接続を試行
    const result = await page.evaluate(async () => {
      return new Promise<string>((resolve) => {
        const ws = new WebSocket(`ws://${location.host}/ws/terminal?sessionId=test`);
        ws.onopen = () => resolve("connected");
        ws.onerror = () => resolve("error");
        ws.onclose = () => resolve("closed");
        setTimeout(() => resolve("timeout"), 5000);
      });
    });

    expect(result).not.toBe("connected");
  });

  // E2E-10: 誤認証 WS 接続拒否
  test("誤った認証情報で WebSocket 接続が拒否される", async ({ page }) => {
    // 誤った認証情報での接続を試行
    // ブラウザ標準 WebSocket API ではカスタムヘッダーを設定できないため、
    // 認証されていないコンテキストでの接続テスト
    const result = await page.evaluate(async () => {
      return new Promise<string>((resolve) => {
        const ws = new WebSocket(`ws://${location.host}/ws/terminal?sessionId=test`);
        ws.onopen = () => resolve("connected");
        ws.onerror = () => resolve("error");
        ws.onclose = () => resolve("closed");
        setTimeout(() => resolve("timeout"), 5000);
      });
    });

    expect(result).not.toBe("connected");
  });

  // E2E-11: 認証後のみ入力可能
  test("認証済みの場合のみキー入力が tmux pane に反映される", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await terminalBtn.click();

    const terminal = page.locator(".xterm");
    await expect(terminal).toBeVisible();

    // 入力テスト
    await terminal.click();
    await page.keyboard.type("echo authenticated");
    await page.keyboard.press("Enter");

    await page.waitForTimeout(3000);
    const screen = page.locator(".xterm-screen");
    const text = await screen.textContent();
    expect(text).toContain("authenticated");
  });
});

test.describe("ターミナルビューア: リサイズ", () => {
  // E2E-12: ターミナルリサイズ
  test("ブラウザウィンドウサイズ変更でターミナルがリサイズされる", async ({ page }) => {
    await page.goto("/");
    await page.waitForSelector("[data-testid='active-sessions']");

    const terminalBtn = page.locator("button[title='ターミナルを開く']").first();
    await terminalBtn.click();

    const terminal = page.locator(".xterm");
    await expect(terminal).toBeVisible();

    // ウィンドウサイズ変更
    await page.setViewportSize({ width: 1200, height: 800 });
    await page.waitForTimeout(1000);

    // ターミナル領域がリサイズされていること
    const terminalBox = await terminal.boundingBox();
    expect(terminalBox).toBeTruthy();
    expect(terminalBox!.width).toBeGreaterThan(400);
  });
});
```

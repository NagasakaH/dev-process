## 作業内容

### 目的

staticメソッドやDI外のコードで手動トレースを実現する`TraceHelper`静的クラスを実装する。これにより、DIを使用しないコードでもトレースが可能になる。

### 設計参照

- [dev-design/02_interface-api-design.md](../dev-design/02_interface-api-design.md) - 3.1 TraceHelper クラス
- [dev-design/04_processing-flow-design.md](../dev-design/04_processing-flow-design.md) - 4.1 TraceHelper.StartTrace

### 実装ステップ

1. **TraceHelper 静的クラス作成**
   - DefaultActivitySource プロパティ
   - StartTrace メソッド（基本版、親コンテキスト指定版）
   - WrapAsync メソッド（Task、Task<T>）
   - Wrap メソッド（同期版）
   - SetTag, AddEvent, RecordException メソッド

2. **StartLinkedTrace 実装**
   - Fire-and-Forget用のリンク関係トレース

3. **単体テスト実装**
   - 各メソッドの動作確認
   - 親子関係の確認

```csharp
namespace TracingSample.Tracing.Tests.Unit;

public class TraceHelperTests : IDisposable
{
    private readonly ActivitySource _activitySource;
    private readonly ActivityListener _listener;
    private readonly List<Activity> _activities = new();

    public TraceHelperTests()
    {
        _activitySource = new ActivitySource("TraceHelperTest");
        TraceHelper.DefaultActivitySource = _activitySource;
        
        _listener = new ActivityListener
        {
            ShouldListenTo = _ => true,
            Sample = (ref ActivityCreationOptions<ActivityContext> _) => ActivitySamplingResult.AllDataAndRecorded,
            ActivityStarted = a => _activities.Add(a)
        };
        ActivitySource.AddActivityListener(_listener);
    }

    public void Dispose()
    {
        _listener.Dispose();
        _activitySource.Dispose();
        TraceHelper.DefaultActivitySource = null;
    }

    [Fact]
    public void StartTrace_CreatesAndDisposesActivity()
    {
        // Act
        using (TraceHelper.StartTrace("TestOperation"))
        {
            Assert.NotNull(Activity.Current);
            Assert.Equal("TestOperation", Activity.Current.DisplayName);
        }

        // Assert
        Assert.Single(_activities);
    }

    [Fact]
    public void StartTrace_WithParentContext_CreatesChildActivity()
    {
        // Arrange
        ActivityContext parentContext;
        using (var parent = _activitySource.StartActivity("Parent"))
        {
            parentContext = parent!.Context;
        }

        // Act
        using (TraceHelper.StartTrace("Child", parentContext))
        {
            // Assert
            Assert.Equal(parentContext.TraceId, Activity.Current?.Context.TraceId);
        }
    }

    [Fact]
    public void StartTrace_WithoutActivitySource_ReturnsNoOpScope()
    {
        // Arrange
        TraceHelper.DefaultActivitySource = null;

        // Act
        var scope = TraceHelper.StartTrace("Test");

        // Assert
        Assert.NotNull(scope);
        scope.Dispose(); // Should not throw

        {
            await Task.Delay(1);

        });

        // Assert
        Assert.Equal(42, result);

    public async Task WrapAsync_WithException_RecordsExceptionAndRethrows()

            await TraceHelper.WrapAsync("FailingOp", async () =>
```

```csharp
using System.Diagnostics;
using TracingSample.Tracing.Internal;

namespace TracingSample.Tracing.Helpers;

/// <summary>
/// 手動トレース作成用のヘルパークラス。
/// DIを使用しないstaticメソッドやユーティリティで使用します。
/// </summary>
public static class TraceHelper
{
    /// <summary>
    /// デフォルトのActivitySourceを取得または設定します。
    /// </summary>
    public static ActivitySource? DefaultActivitySource { get; set; }

    /// <summary>
    /// トレースを開始します。
    /// </summary>
    public static IDisposable StartTrace(
        string name,
        ActivityKind kind = ActivityKind.Internal)
    {
        if (DefaultActivitySource == null)
        {
            return NoOpScope.Instance;
        }

        var activity = DefaultActivitySource.StartActivity(name, kind);
        return new TraceScope(activity, Activity.Current?.Context, TracingOptions.Default);
    }

    /// <summary>
    /// 親コンテキストを指定してトレースを開始します。
    /// </summary>
    public static IDisposable StartTrace(
        string name,
        ActivityContext parentContext,
        ActivityKind kind = ActivityKind.Internal)
    {
        if (DefaultActivitySource == null)
        {
            return NoOpScope.Instance;
        }

        var activity = DefaultActivitySource.StartActivity(name, kind, parentContext);
        return new TraceScope(activity, parentContext, TracingOptions.Default);
    }

    /// <summary>
    /// Fire-and-Forget用のトレース開始（リンク関係）
    /// </summary>
    public static IDisposable StartLinkedTrace(
        string name,
        ActivityContext linkedContext,
        ActivityKind kind = ActivityKind.Internal)
    {
        if (DefaultActivitySource == null)
        {
            return NoOpScope.Instance;
        }

        var links = new[] { new ActivityLink(linkedContext) };
        var activity = DefaultActivitySource.StartActivity(
            name,
            kind,
            default(ActivityContext),
            null,
            links);

        return new TraceScope(activity, null, TracingOptions.Default);
    }

    /// <summary>
    /// トレースをラップした非同期処理を実行します。
    /// </summary>
    public static async Task<T> WrapAsync<T>(string name, Func<Task<T>> func)
    {
        using var scope = StartTrace(name) as TraceScope;
        try
        {
            var result = await func();
            scope?.SetSuccess();
            return result;
        }
        catch (Exception ex)
        {
            scope?.RecordException(ex);
            throw;
        }
    }

    /// <summary>
    /// トレースをラップした非同期処理を実行します（戻り値なし）。
    /// </summary>
    public static async Task WrapAsync(string name, Func<Task> action)
    {
        using var scope = StartTrace(name) as TraceScope;
        try
        {
            await action();
            scope?.SetSuccess();
        }
        catch (Exception ex)
        {
            scope?.RecordException(ex);
            throw;
        }
    }

    /// <summary>
    /// トレースをラップした同期処理を実行します。
    /// </summary>
    public static T Wrap<T>(string name, Func<T> func)
    {
        using var scope = StartTrace(name) as TraceScope;
        try
        {
            var result = func();
            scope?.SetSuccess();
            return result;
        }
        catch (Exception ex)
        {
            scope?.RecordException(ex);
            throw;
        }
    }

    /// <summary>
    /// トレースをラップした同期処理を実行します（戻り値なし）。
    /// </summary>
    public static void Wrap(string name, Action action)
    {
        using var scope = StartTrace(name) as TraceScope;
        try
        {
            action();
            scope?.SetSuccess();
        }
        catch (Exception ex)
        {
            scope?.RecordException(ex);
            throw;
        }
    }

    /// <summary>
    /// 現在のActivityにタグを追加します。
    /// </summary>
    public static void SetTag(string key, object? value)
    {
        Activity.Current?.SetTag(key, value);
    }

    /// <summary>
    /// 現在のActivityにイベントを追加します。
    /// </summary>
    public static void AddEvent(string name, IDictionary<string, object?>? tags = null)
    {
        if (Activity.Current != null)
        {
            var eventTags = tags != null
                ? new ActivityTagsCollection(
                    tags.Select(kv => new KeyValuePair<string, object?>(kv.Key, kv.Value)))
                : null;
            Activity.Current.AddEvent(new ActivityEvent(name, DateTimeOffset.UtcNow, eventTags));
        }
    }

    /// <summary>
    /// 現在のActivityに例外を記録します。
    /// </summary>
    public static void RecordException(Exception exception)
    {
        if (Activity.Current != null)
        {
            Activity.Current.SetTag("exception.type", exception.GetType().FullName);
            Activity.Current.SetTag("exception.message", exception.Message);
            Activity.Current.SetTag("exception.stacktrace", exception.StackTrace);
            Activity.Current.SetStatus(ActivityStatusCode.Error, exception.Message);
        }
    }
}
```
